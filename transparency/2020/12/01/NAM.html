Neural Networks as Interpretable as Linear Models

The classic Neural Network (NN) with many layers and non-linear activation functions is difficult to interpret. However, that doesn't mean all NNs are opaque. This post delves into the history of General Linear Models (GLMs), and shows that with intentional design, NNs can be just as transparent.

import tensorflow as tf
import matplotlib.pyplot as plt
import pandas as pd
import numpy as np

columns = ['a','b','c']
n_data = 100000
mean = 0.0
std = 1.0
#create the dataset
data = {
    col: np.random.normal(
        size=n_data, loc=mean, scale=std
    ) for col in columns
}
df = pd.DataFrame(data)
#apply nonlinear transforms to the data
functions = {
    'a': lambda x: 5 * (2 ** x),
    'b': lambda x: x**4 + x**3 - 10 * x**2 + x,
    'c': lambda x: 10 / ((abs(x) ** 0.5) + 1e-1),
}
for col in columns:
    df[f'{col}_transform'] = df[col].apply(functions[col])
#set up a prediction task where the target is a composition of
#    the nonlinear transforms, but the training data is the normally
#    distributed data prior to being transformed
X = df[[c for c in df.columns if not c.endswith('_transform')]]
y = df[[c for c in df.columns if c.endswith('_transform')]].sum(1)

class MLP(tf.keras.Model):
    def __init__(
        self,
        name=None
    ):
        super().__init__(name=name)
        self.layer1 = tf.keras.layers.Dense(8, activation='softplus')
        self.layer2 = tf.keras.layers.Dense(8, activation='softplus')
        self.layer3 = tf.keras.layers.Dense(8, activation='softplus')
        self.output_layer = tf.keras.layers.Dense(1)
    
    def call(self, x, training=None):
        x = self.layer1(x)
        x = self.layer2(x)
        x = self.layer3(x)
        return self.output_layer(x)
    
#define a Neural Additive Model for n features
class NAM(tf.keras.Model):
    def __init__(
        self,
        n_features,
        name=None
    ):
        super().__init__(name=name)
        self.n_features = n_features
        self.components = [MLP() for i in range(n_features)]
    
    @tf.function
    def call(self, x, training=None):
        individual_features = tf.split(x, self.n_features, axis=1)
        components = []
        for f_idx,individual_feature in enumerate(individual_features):
            component = self.components[f_idx](individual_feature)
            components.append(component)
        return tf.add_n(components)

model = NAM(len(X.columns))
model.compile(
    optimizer=tf.keras.optimizers.Adam(),
    loss='mean_squared_error',
    metrics=['mean_absolute_error']
)
model.fit(
    X.to_numpy(),
    y.to_numpy(),
    epochs=50,
    batch_size=32,
)
